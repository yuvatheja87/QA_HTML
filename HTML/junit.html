Annotations

	@BeforeClass	<-- executes once before all the tests
	public static void beforeclass(){
	}
	@Before			<--executes before every test
	public void setup(){
	}
	@Test
	public void test_gvjhvjhb_conditionname(){
	}
	@After			<--executes after every test
	public void teardown(){
	}
	@AfterClass		<--executes once after all the tests
	public static void afterclass(){
	}
